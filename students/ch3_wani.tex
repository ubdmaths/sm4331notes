\subsection{Definition}\label{definition}

\begin{definition}[]\protect\hypertarget{def-line}{}\label{def-line}

Let \(X_1 \sim X_{k1}^2\). Then, the distribution of

\[
Y = \frac{X_1 / k_1}{X_2 / k_2}
\]

is called the \(F\)-distribution with (\(k_1,k_2\)) degrees of freedom.
We write \(Y \sim F_{k_1,k_2}\)

\end{definition}

\subsection{Probability Density Function
(PDF)}\label{probability-density-function-pdf}

\begin{tcolorbox}[enhanced jigsaw, coltitle=black, bottomtitle=1mm, rightrule=.15mm, opacityback=0, opacitybacktitle=0.6, bottomrule=.15mm, leftrule=.75mm, breakable, colframe=quarto-callout-note-color-frame, toptitle=1mm, titlerule=0mm, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Note}, arc=.35mm, toprule=.15mm, left=2mm, colback=white, colbacktitle=quarto-callout-note-color!10!white]

\textbf{PDF:}

\[
f(y) = \frac{1}{yB(k_1/2, k_2/2)}\sqrt{\frac{(k_1y)^{k_1}k_2 ^{k_2}}{(k_1y + k_2)^{k_1 + k_2}}} \hspace{10mm} \left({y \in \mathbb{R} \mid y>0} \right)
\]

\end{tcolorbox}
